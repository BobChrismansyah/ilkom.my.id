#include <iostream>
#include <vector>

int main() {
    std::vector<char> myStack;
    std::string data;
    std::getline(std::cin, data);

    for(char & i : data) {
        if(i == '(') {
            myStack.push_back(i);
        } else if (i == ')' && !myStack.empty()) {
            myStack.pop_back();
        } else if (i == ')' && myStack.empty()) {
            std::cout << "Not fix" << std::endl;
            break;
        }
    }

    std::cout << (myStack.empty() ? "All data correct" : "Not fix") << std::endl;
    return 0;
}
